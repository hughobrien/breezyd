# Basic UI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers-extended-cc:subagent-driven-development (recommended) or superpowers-extended-cc:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a single-page three-column dashboard served by `breezyd` itself at `GET /`, embedded into the binary via `go:embed`. One card per configured device, polling `GET /v1/devices/<name>` every 5s, with controls for power/mode/speed/heater that POST to existing handlers. Plus an opt-in NixOS module integration that fronts the daemon with nginx (with optional basic auth) so the dashboard can be exposed on the LAN without binding the daemon's full API off-loopback.

**Architecture:** New `cmd/breezyd/ui.go` holds a single `getIndex` handler that serves the embedded HTML. New `cmd/breezyd/ui/index.html` contains the entire UI (HTML + inlined `<style>` + inlined `<script>`). `cmd/breezyd/server.go` gains one route registration: `mux.HandleFunc("GET /{$}", h.getIndex)` — note the `{$}` end-of-path anchor so the handler matches **only** the root path, not every unmatched URL. JSON shape consumed is the existing `SnapshotResponse` from `cmd/breezyd/snapshot.go`. No new HTTP routes for device data; controls POST to the existing `/v1/devices/{name}/{power,mode,speed,heater}` endpoints. The NixOS module gains a `services.breezyd.nginx` option block that mirrors the existing `prometheus` integration shape.

**Tech Stack:** Go 1.22 `embed`, `net/http` `ServeMux` 1.22 patterns, vanilla HTML/CSS/JS — no framework, no build step, no CDN. NixOS module options + `services.nginx.virtualHosts` integration.

**Spec:** `docs/superpowers/specs/2026-05-04-basic-ui-design.md`

```html
  <header>
    <h1>breezy</h1>
    <span class="meta" id="meta">loading…</span>
  </header>
  <div id="banner"></div>
  <div class="grid" id="grid"></div>
<script>
"use strict";

const POLL_INTERVAL_MS = 5000;
const STALE_THRESHOLD_MS = 90 * 1000;

let deviceNames = [];
const lastSnapshots = {}; // name -> {snapshot fields, _fetchedAt}

async function bootstrap() {
  try {
    const r = await fetch("/v1/devices");
    if (!r.ok) throw new Error(`HTTP ${r.status}`);
    const data = await r.json();
    deviceNames = (data.devices || []).map(d => d.name).sort();
    clearBanner();
    render();
    await refreshAll();
    setInterval(refreshAll, POLL_INTERVAL_MS);
    setInterval(updateMeta, 1000);
  } catch (e) {
    showBanner(`cannot reach daemon — is breezyd running? (${e.message})`);
    setTimeout(bootstrap, POLL_INTERVAL_MS);
  }
}

function showBanner(msg) {
  document.getElementById("banner").innerHTML =
    `<div class="err-banner">${esc(msg)}</div>`;
}
function clearBanner() {
  document.getElementById("banner").innerHTML = "";
}

async function refreshAll() {
  const results = await Promise.allSettled(
    deviceNames.map(n =>
      fetch(`/v1/devices/${encodeURIComponent(n)}`)
        .then(r => r.ok ? r.json() : Promise.reject(new Error(`HTTP ${r.status}`)))
        .then(snap => ({ name: n, snap }))
    )
  );
  for (const res of results) {
    if (res.status === "fulfilled") {
      lastSnapshots[res.value.name] = { ...res.value.snap, _fetchedAt: Date.now() };
    }
  }
  render();
}

function render() {
  document.getElementById("grid").innerHTML =
    deviceNames.map(name => renderCard(name, lastSnapshots[name])).join("");
  updateMeta();
}

function updateMeta() {
  const meta = document.getElementById("meta");
  const last = Math.max(0, ...Object.values(lastSnapshots).map(s => s._fetchedAt || 0));
  meta.textContent = last === 0 ? "no data yet" : `refreshed ${humanAgo(Date.now() - last)} ago`;
}

function renderCard(name, snap) {
  if (!snap) {
    return `<div class="card stale">
      <h2>${esc(name)}</h2>
      <div class="ts">no data</div>
    </div>`;
  }
  const lastPollMs = snap.last_poll ? new Date(snap.last_poll).getTime() : 0;
  const ageMs = lastPollMs ? Date.now() - lastPollMs : Infinity;
  const stale = ageMs > STALE_THRESHOLD_MS;
  const power = snap.configured?.power;
  const powerLabel = power === true ? "on" : power === false ? "off" : "?";
  const powerCls  = power === true ? "power-on" : "power-off";

  return `<div class="card${stale ? " stale" : ""}">
    <div class="row">
      <h2>${esc(name)}</h2>
      <span class="${powerCls}">● ${powerLabel}</span>
    </div>
    <div class="row">
      <span class="ip">${esc(snap.ip || "?")}</span>
      <span class="ts${stale ? " red" : ""}">${lastPollMs ? humanAgo(ageMs) + " ago" : "no poll"}</span>
    </div>

    <div class="block">
      <h3>Sensors</h3>
      ${kv("RH",       fmt(snap.sensors?.humidity_pct, "—", "%"))}
      ${kv("eCO₂",     fmt(snap.sensors?.eco2_ppm, "—", " ppm"))}
      ${kv("VOC",      fmt(snap.sensors?.voc_index, "—"))}
      ${kv("outdoor",  tempC(snap.sensors?.temp_outdoor_c))}
      ${kv("supply",   tempC(snap.sensors?.temp_supply_c))}
      ${kv("recovery", fmt(snap.sensors?.recovery_efficiency_pct, "—", "%"))}
    </div>

    <div class="block">
      <h3>Fans</h3>
      ${kv("supply",   fmt(snap.live?.fan_supply_rpm, "—", " rpm"))}
      ${kv("extract",  fmt(snap.live?.fan_extract_rpm, "—", " rpm"))}
      ${overrideLine(snap.live)}
    </div>

    <div class="block">
      <h3>Service</h3>
      ${kv("filter",  filterStatus(snap.service))}
      ${kv("motor",   humanSeconds(snap.service?.motor_lifetime_seconds))}
      ${kv("RTC",     fmt(snap.service?.rtc_battery_volts, "—", " V"))}
      ${kv("faults",  fmt(snap.service?.fault_level, "—"))}
    </div>

    ${snap.firmware ? `<div class="fw">fw ${esc(snap.firmware.version || "?")} · ${esc(snap.firmware.build_date || "?")}</div>` : ""}
  </div>`;
}

function kv(k, v) {
  return `<div class="row"><span class="k">${esc(k)}</span><span>${v}</span></div>`;
}

function tempC(v) {
  return (v === null || v === undefined) ? "—" : `${v.toFixed(1)} °C`;
}

function fmt(v, fallback, suffix) {
  if (v === null || v === undefined) return fallback;
  return `${esc(String(v))}${suffix || ""}`;
}

function filterStatus(svc) {
  if (!svc) return "—";
  const status = svc.filter_status || "—";
  const remaining = svc.filter_remaining_seconds;
  if (remaining === undefined) return esc(status);
  const days = Math.floor(remaining / 86400);
  return `${esc(status)} (${days}d)`;
}

function overrideLine(live) {
  if (!live || live.in_user_control !== false) return "";
  const alerts = live.sensor_alerts || {};
  const reasons = ["humidity", "co2", "voc"].filter(k => alerts[k]);
  const reasonText = reasons.length ? ` (${reasons.join(", ")})` : "";
  return `<div class="warn">⚠ sensor override${reasonText} — fan above setting</div>`;
}

function humanAgo(ms) {
  const s = Math.max(0, Math.floor(ms / 1000));
  if (s < 60) return `${s}s`;
  const m = Math.floor(s / 60);
  if (m < 60) return `${m}m ${s % 60}s`;
  const h = Math.floor(m / 60);
  return `${h}h ${m % 60}m`;
}

function humanSeconds(secs) {
  if (secs === null || secs === undefined) return "—";
  const h = Math.floor(secs / 3600);
  const m = Math.floor((secs % 3600) / 60);
  return `${h}h ${m}m`;
}

function esc(s) {
  return String(s ?? "").replace(/[&<>"']/g, c => ({
    "&": "&amp;", "<": "&lt;", ">": "&gt;", '"': "&quot;", "'": "&#39;"
  }[c]));
}

bootstrap();
</script>
</body>
</html>
```

- [ ] **Step 2: Create the handler at `cmd/breezyd/ui.go`**

```go
// SPDX-License-Identifier: GPL-3.0-or-later

// HTTP handler for the embedded single-page dashboard. The UI is a single
// self-contained HTML file (CSS + JS inlined) that lives at
// cmd/breezyd/ui/index.html and ships baked into the binary via go:embed.
//
// The handler is intentionally minimal: one route (GET /), one byte slice,
// no caching, no asset tree. If the UI grows past one file, this is the
// place to switch to embed.FS + http.FileServerFS.
package main

import (
	_ "embed"
	"net/http"
)

//go:embed ui/index.html
var indexHTML []byte

// getIndex serves the embedded dashboard HTML. We send Cache-Control:
// no-store so an upgraded daemon's UI is picked up on the next page load
// rather than after a hard refresh.
func (h *Handler) getIndex(w http.ResponseWriter, r *http.Request) {
	w.Header().Set("Content-Type", "text/html; charset=utf-8")
	w.Header().Set("Cache-Control", "no-store")
	_, _ = w.Write(indexHTML)
}
```

- [ ] **Step 3a: Wire the route on the inner mux in `cmd/breezyd/server.go`**

In `Handler.routes()`, add the index registration. Use the `GET /{$}` form so the handler matches **only** the literal root path — `GET /` (no trailing-slash anchor) would catch every unmatched URL and silently turn API typos into HTML responses.

Find this block in `cmd/breezyd/server.go` (around line 184–205):

```go
func (h *Handler) routes() *http.ServeMux {
	mux := http.NewServeMux()

	mux.HandleFunc("GET /healthz", h.healthz)
	mux.HandleFunc("GET /v1/devices", h.listDevices)
	mux.HandleFunc("GET /v1/devices/{name}", h.getDevice)
	...
```

Add the new line above `GET /healthz`:

```go
func (h *Handler) routes() *http.ServeMux {
	mux := http.NewServeMux()

	mux.HandleFunc("GET /{$}", h.getIndex)
	mux.HandleFunc("GET /healthz", h.healthz)
	mux.HandleFunc("GET /v1/devices", h.listDevices)
	mux.HandleFunc("GET /v1/devices/{name}", h.getDevice)
	...
```

(Position is cosmetic; keep the existing routes in their original order.)

- [ ] **Step 3b: Route the root path through the outer mux in `cmd/breezyd/main.go`**

The daemon wires a small outer `http.ServeMux` in `cmd/breezyd/main.go` (around line 138–141) that fans out to two different handlers — `Handler` for the API and a separate `metricsHandler` for Prometheus:

```go
mux := http.NewServeMux()
mux.Handle("/healthz", handler)
mux.Handle("/v1/", handler)
mux.Handle("/metrics", metricsHandler(reg, metrics, state, devices))
```

The outer mux currently has nothing routed for `/`, so a request to `GET /` would 404 at the outer layer **before** it could ever reach the `Handler`'s inner mux (where the new `GET /{$}` lives). Add a catch-all `/` registration that forwards to `handler`:

```go
mux := http.NewServeMux()
mux.Handle("/healthz", handler)
mux.Handle("/v1/", handler)
mux.Handle("/metrics", metricsHandler(reg, metrics, state, devices))
mux.Handle("/", handler)
```

Notes on the precedence:
- `/metrics` is an exact-match registration; it still wins over the `/` prefix.
- The existing `/healthz` and `/v1/` registrations are now technically redundant (the new `/` prefix would catch them too, and `handler.routes()` matches them either way), but leave them — explicit is fine, and the diff stays minimal.
- The `Handler`'s inner mux uses `GET /{$}` (Go 1.22 end-of-path anchor), so a typo like `GET /asdf` reaches `handler` via the outer mux but no inner route matches → 404. The third smoke test in Step 4 covers this regression.

- [ ] **Step 4: Add tests at `cmd/breezyd/ui_test.go`**

These are smoke tests that verify the handler is wired correctly and doesn't break the API. The HTML/JS itself is not unit-tested — its data shape is the existing `SnapshotResponse`, which has its own coverage in `server_test.go`.

```go
// SPDX-License-Identifier: GPL-3.0-or-later

package main

import (
	"net/http"
	"strings"
	"testing"
)

// TestUI_GetIndex confirms GET / returns the embedded dashboard with the
// correct Content-Type and a non-empty body that includes the page title.
func TestUI_GetIndex(t *testing.T) {
	h, _, _ := newServerHandler(t)
	rec := doRequest(t, h, http.MethodGet, "/", nil)

	if rec.Code != http.StatusOK {
		t.Fatalf("status = %d, want 200", rec.Code)
	}
	if got := rec.Header().Get("Content-Type"); got != "text/html; charset=utf-8" {
		t.Errorf("Content-Type = %q, want text/html; charset=utf-8", got)
	}
	if got := rec.Header().Get("Cache-Control"); got != "no-store" {
		t.Errorf("Cache-Control = %q, want no-store", got)
	}
	body := rec.Body.String()
	if len(body) == 0 {
		t.Fatal("body is empty; go:embed likely not wired")
	}
	if !strings.Contains(body, "<title>breezy</title>") {
		t.Errorf("body missing <title>breezy</title>; got prefix %q", body[:min(200, len(body))])
	}
}

// TestUI_DoesNotInterceptAPI is a regression: the new GET /{$} pattern
// must not catch other API paths. /v1/devices should still return JSON.
func TestUI_DoesNotInterceptAPI(t *testing.T) {
	h, _, _ := newServerHandler(t)
	rec := doRequest(t, h, http.MethodGet, "/v1/devices", nil)

	if rec.Code != http.StatusOK {
		t.Fatalf("status = %d, want 200", rec.Code)
	}
	if got := rec.Header().Get("Content-Type"); !strings.HasPrefix(got, "application/json") {
		t.Errorf("Content-Type = %q, want application/json...", got)
	}
}

// TestUI_UnknownPath confirms the index pattern does NOT catch arbitrary
// unmatched paths. A typo on the API surface should 404, not return HTML.
func TestUI_UnknownPath(t *testing.T) {
	h, _, _ := newServerHandler(t)
	rec := doRequest(t, h, http.MethodGet, "/asdf", nil)

	if rec.Code != http.StatusNotFound {
		t.Fatalf("status = %d, want 404", rec.Code)
	}
}
```

### Task 2: Per-card controls (power, mode, speed, heater)

**Goal:** Add the four high-level controls to each card. Click a control → POST to the existing handler → wait for the next poll to see the updated state. POST failures show an inline toast under the failing control with the daemon's error envelope text. Controls disable while a write is in flight or while the card is stale.

**Files:**
- Modify: `cmd/breezyd/ui/index.html` (add CSS for controls, control HTML in each card, JS click handlers + toast logic)

**Acceptance Criteria:**
- [ ] Each card has a Controls block with: a Power toggle button, four Mode buttons (ventilation/regeneration/supply/extract), a Speed segmented control (preset 1/2/3 + manual), a manual % slider (10–100, step 5), and a Heater toggle button.
- [ ] Clicking Power POSTs `{"on": <inverse-of-current>}` to `/v1/devices/<name>/power`.
- [ ] Clicking a Mode button POSTs `{"mode": "<button-label>"}` to `/v1/devices/<name>/mode`.
- [ ] Clicking a preset radio POSTs `{"preset": <1|2|3>}` to `/v1/devices/<name>/speed` immediately.
- [ ] Clicking the manual radio enables the slider but does NOT fire a POST until the slider is moved.
- [ ] The slider fires `POST /v1/devices/<name>/speed` with `{"manual": <pct>}` on `change` (mouseup/blur), not `input` (every drag tick).
- [ ] Clicking Heater POSTs `{"on": <inverse-of-current>}` to `/v1/devices/<name>/heater`.
- [ ] On POST failure, a `.toast` div appears under the failing control with the daemon's `error` field text. Toast auto-clears on the next successful poll for that device.
- [ ] All controls are `disabled` (and visually dimmed) while a write is in flight for that device, and while the card is stale (`ageMs > STALE_THRESHOLD_MS`).
- [ ] The control widgets reflect `configured.*` values (what the user asked for), NOT `live.*` values. Speed mode shows `preset N` or `manual N%` based on `configured.speed_mode` and `configured.manual_pct`.

```javascript
async function postWrite(name, controlKey, url, body) {
  inFlight[name] = true;
  render();
  try {
    const r = await fetch(url, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify(body),
    });
    if (!r.ok) {
      let msg = `HTTP ${r.status}`;
      try {
        const env = await r.json();
        if (env.error) msg = env.error;
      } catch {}
      setToast(name, controlKey, msg);
    } else {
      clearToast(name, controlKey);
      // Refresh this device immediately so the user sees feedback before
      // the next 5s tick. Other devices are unaffected.
      try {
        const snapResp = await fetch("/v1/devices/" + encodeURIComponent(name));
        if (snapResp.ok) {
          lastSnapshots[name] = { ...(await snapResp.json()), _fetchedAt: Date.now() };
        }
      } catch {}
    }
  } catch (e) {
    setToast(name, controlKey, e.message);
  } finally {
    inFlight[name] = false;
    render();
  }
}

function setToast(name, controlKey, msg) {
  if (!toasts[name]) toasts[name] = {};
  toasts[name][controlKey] = msg;
}
function clearToast(name, controlKey) {
  if (toasts[name]) delete toasts[name][controlKey];
}
```

```javascript
  const results = await Promise.allSettled(
    deviceNames.map(n =>
      fetch(`/v1/devices/${encodeURIComponent(n)}`)
        .then(r => r.ok ? r.json() : Promise.reject(new Error(`HTTP ${r.status}`)))
```

```typescript
// loadDashboard wires up route mocks and serves the embedded HTML to the
// browser via setContent. Returns a record of intercepted POSTs that the
// test can assert against.
type RecordedRequest = { url: string; method: string; body: any };

async function loadDashboard(
  page: Page,
  opts: {
    devices?: { name: string }[];
    snapshot?: (name: string) => any;
    postResponse?: (req: { url: string; method: string; body: any }) => {
      status: number;
      body: any;
    };
    failBootstrap?: boolean;
  } = {},
): Promise<{ requests: RecordedRequest[] }> {
  const devices = opts.devices ?? [{ name: "playroom" }, { name: "bedroom" }];
  const snapshot = opts.snapshot ?? ((n) => baseSnapshot(n));
  const requests: RecordedRequest[] = [];

  await page.route("**/v1/devices", async (route: Route) => {
    if (opts.failBootstrap) {
      await route.fulfill({ status: 502, body: "bad gateway" });
      return;
    }
    await route.fulfill({
      status: 200,
      contentType: "application/json",
      body: JSON.stringify({ devices }),
    });
  });

  await page.route("**/v1/devices/*", async (route: Route) => {
    const req = route.request();
    const url = req.url();
    const method = req.method();
    let body: any = null;
    try { body = JSON.parse(req.postData() ?? ""); } catch {}
    requests.push({ url, method, body });

    if (method === "GET") {
      const name = decodeURIComponent(url.split("/").pop() ?? "");
      await route.fulfill({
        status: 200,
        contentType: "application/json",
        body: JSON.stringify(snapshot(name)),
      });
      return;
    }

    if (method === "POST") {
      const resp = opts.postResponse?.({ url, method, body });
      await route.fulfill({
        status: resp?.status ?? 200,
        contentType: "application/json",
        body: JSON.stringify(resp?.body ?? { ok: true }),
      });
      return;
    }

    await route.continue();
  });

  // Catches the deeper /v1/devices/<name>/<sub>... POSTs (power, mode, etc.)
  await page.route("**/v1/devices/*/*", async (route: Route) => {
    const req = route.request();
    const url = req.url();
    const method = req.method();
    let body: any = null;
    try { body = JSON.parse(req.postData() ?? ""); } catch {}
    requests.push({ url, method, body });

    const resp = opts.postResponse?.({ url, method, body });
    await route.fulfill({
      status: resp?.status ?? 200,
      contentType: "application/json",
      body: JSON.stringify(resp?.body ?? { ok: true }),
    });
  });

  // setContent does not give us a useful base URL, but page.route() matches
  // by URL pattern regardless of origin, so fetch("/v1/devices") hitting
  // about:blank is intercepted before the browser ever tries the network.
  await page.setContent(INDEX_HTML);
  return { requests };
}
```

```go
// ErrConfigExists is returned by WriteDefault when the target path
// already exists. Callers can use errors.Is to distinguish this from
// other write failures and avoid clobbering a real config.
var ErrConfigExists = errors.New("config: file already exists")

// WriteDefault writes a starter config to path with mode 0600. The
// parent directory is created (mode 0700) if missing. The write is
// atomic: content is staged in a sibling tempfile and renamed into
// place, so a crash mid-write doesn't leave a half-formed file that
// the loader would reject.
//
// If path already exists, WriteDefault returns ErrConfigExists without
// touching the existing file. Callers that want to overwrite must
// handle that themselves.
func WriteDefault(path string) error {
	if _, err := os.Stat(path); err == nil {
		return fmt.Errorf("%w: %s", ErrConfigExists, path)
	} else if !errors.Is(err, os.ErrNotExist) {
		return fmt.Errorf("config: stat %s: %w", path, err)
	}

	dir := filepath.Dir(path)
	if dir != "" && dir != "." {
		if err := os.MkdirAll(dir, 0o700); err != nil {
			return fmt.Errorf("config: mkdir %s: %w", dir, err)
		}
	}

	// Stage in a sibling temp file with the same dir so rename is atomic.
	tmp, err := os.CreateTemp(dir, ".breezyd-config-*.tmp")
	if err != nil {
		return fmt.Errorf("config: create temp: %w", err)
	}
	tmpPath := tmp.Name()
	// On any failure between here and Rename, remove the temp file.
	cleanup := func() { _ = os.Remove(tmpPath) }

	if _, err := tmp.WriteString(defaultConfigTemplate); err != nil {
		_ = tmp.Close()
		cleanup()
		return fmt.Errorf("config: write temp: %w", err)
	}
	if err := tmp.Chmod(0o600); err != nil {
		_ = tmp.Close()
		cleanup()
		return fmt.Errorf("config: chmod temp: %w", err)
	}
	if err := tmp.Close(); err != nil {
		cleanup()
		return fmt.Errorf("config: close temp: %w", err)
	}
	if err := os.Rename(tmpPath, path); err != nil {
		cleanup()
		return fmt.Errorf("config: rename temp -> %s: %w", path, err)
	}
	return nil
}
```
